@variable(m, x[source, sink] >= 0)

arcs_add3 = []

    if i - 1 >= 6
        append!(arcs_sub1,[(i,i-1)])
    end
    if i * 2 <= 2024
        append!(arcs_mul2,[(i, i * 2)])
    end
    if i + 3 <= 2024
        append!(arcs_add3,[(i, i + 3)])
    end
